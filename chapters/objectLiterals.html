Object Literals

Object Literal in ES6

Object Literal in JS

Syntax

var obj={
            key:value,
            ...
            ...
          }

Example

var pData={
name:"Jack the Ripper",
age:40,
kids:['Andy Nukem','Crazy Lo']
}
Object Literals

Pre ES6


function toObj(name,age,kids){
//Notice the repetition
var pData={
  name:name,
  age:age,
  kids:kids
};
return pData;
}

ES6


function toObj(name,age,kids){
  const pData={
   name,age,kids
  };
  return pData;
}

const obj=toObj('Jack',40,['Andy','Lo']);
console.log(obj);

Object Literal- Defining functions

Pre Es6

var obj={
  foo:function(){

  },
  bar:function(){

  }
}

Es6

const obj={
    foo(){
      console.log("Foo...");
    },
    bar(){
      console.log("Bar...");
    }
  }
  obj.foo();
  obj.bar();

Object Literals-Compute property names

Pre ES6


          
function toObj(direction,opposite){
            return {
              direction:direction,
              opposite:opposite
            }
          }
        //Make it more descriptive, Change opposite to opposite
        //of left 
          function toObj(direction,opposite){
            var tmpObj={direction:direction};
            tmpObj['oppoiste_of_'+direction]=direction
            return tmpObj;
          }

Object Literals-Compute property names

ES6


          
function toObj(direction,opposite){
  return {
    direction:direction,
    ['oppoiste_of_'+direction]:opposite
  }
}
//Notice property names are computed inside the object


console.log(toObj('left','right'));
//{ direction: "left", oppoiste_of_left: "right" }
