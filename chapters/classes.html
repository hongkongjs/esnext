Classes
JavaScript classes introduced in ECMAScript 2015 are syntactical sugar over JavaScript's existing prototype-based inheritance. The class syntax is not introducing a new object-oriented inheritance model to JavaScript. JavaScript classes provide a much simpler and clearer syntax to create objects and deal with inheritance.

Pre ES6


              
/*Define a function*/
              var House=function(color,
              rooms,floors){
                this.color=color;
                this.rooms=rooms;
                this.floors=floors;
              }
            

            
/*Define descripe function*/
            House.prototype.descripe=
              function(){
                console.log(this.color,
                this.rooms,this.floors);
              }

            var h1=new House("red",2,4);
            console.log(h1);
            //red 2 4

ES6


          class House{
/*Define constructor*/
            constructor(color,rooms,
                        floors) {
             this.color=color;
             this.rooms=rooms;
             this.floors=floors;
           }
/*Define descripe function*/
            descripe(){
                console.log(this.color,
                this.rooms,this.floors);
              }
}
var h1=new House("red",2,4);
          h1.descripe();
          //red 2 4
