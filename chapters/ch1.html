var VS let, const
Scopeing 

              function varBlock() {
                  {
                      var x = 5;
                  }
                  console.log(x);// 5
              }
          

              function letBlock() {
                  {
                      let x = 5;
                  }
                  console.log(x);//undefined
              }
Reassigning 

              function varReassign() {
                  var x = 5;
                  console.log(x); //5
                  x = 1;
                  console.log(x); //1
              }
          

              function letReassign() {
                  let x = 5;
                  console.log(x); //5
                  x = 1;
                  console.log(x); //1
              }
Re-Declaration 

              function varRedeclaration() {
                  var x = 5;
                  console.log(x); // 5
                  var x = 1;
                  console.log(x) //1
              }
          

              function letRedeclaration() {
                  let x = 5;
                  console.log(x); // 5
                  let x = 1;//SyntaxError
                  console.log(x)
              }
Temporal Dead Zone 

              function varTemporalDeadZone() {
                  console.log(x);//undefined
                  var x = 5;
              }
          

              function letTemporalDeadZone() {
                  console.log(x);
                  //Reference Error
                  let x = 5;
              }
