Destructuring
The destructuring assignment syntax is a JavaScript expression that makes it possible to extract data from arrays or objects into distinct variables
Example

Before ES6


//Array
var arr=[10,20];
var arg1=arr[0];
var arg2=arr[1];
//Object
var obj={name:"Superman",
                job:"Pimp"};
var name=obj.name
var job =obj.job

ES6


            //Array
            var arr=[10,20];
            [arg1,arg2]=arr
            //arg1=10, arg2=20
            //Object
            var obj={name:"Superman",
                job:"Pimp"};
            ({name,job}=obj);
            //name=Superman, job=Pimp
Swaping Variables

Before ES6


            var a=10;
            var b=20;
            var tmp=a;
            a=b;
            b=tmp;

ES6


              var a=10;
              var b=20;
              [a,b]=[b,a];
              console.log(a,b); //20 10
Destructing and functions

Before ES6


              var obj={name:"Freddy Krueger",age:9999};
              foo(obj.name,obj.age);

              function foo(name,age){
                //Do something
              }

ES6


              var obj={name:"Freddy Krueger",age:9999};
              foo(obj);

              function foo({name,age}){
                console.log(name,age)
              }
Default Values

Before ES6


              function foo(obj){
                if(obj&&typeof(obj.name)==='undefined'){
                  obj.name="default_value";
                }
              }

ES6


              function foo({name='default_Value'}){
                console.log("name: ",name);
              }

              foo({age:10});
              foo({name:"Valid value"});
